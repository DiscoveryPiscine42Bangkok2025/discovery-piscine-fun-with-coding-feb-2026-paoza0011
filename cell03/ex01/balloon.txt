<!DOCTYPE html>
<html>
<head>
    <title>Balloon</title>

    <style>
        body {
            display: flex;
            justify-content: center;
            align-items: center;
            height: 100vh;
        }

        #balloon {
            width: 200px;
            height: 200px;
            background-color: red;
            border-radius: 50%;
        }
    </style>
</head>

<body>

<div id="balloon"></div>

<script>
let balloon = document.getElementById("balloon");

let size = 200;
let colors = ["red", "green", "blue"];
let index = 0;

/* คลิก = ขยาย + เปลี่ยนสี */
balloon.onclick = function() {
    size += 10;

    if (size > 420) {
        size = 200;
        index = 0;
    } else {
        index = (index + 1) % 3;
    }

    balloon.style.width = size + "px";
    balloon.style.height = size + "px";
    balloon.style.backgroundColor = colors[index];
};

/* เมาส์ออก = หด + สีถอย */
balloon.onmouseleave = function() {
    size -= 5;

    if (size < 200)
        size = 200;

    index--;
    if (index < 0)
        index = 2;

    balloon.style.width = size + "px";
    balloon.style.height = size + "px";
    balloon.style.backgroundColor = colors[index];
};
</script>

</body>
</html>
